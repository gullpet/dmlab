from sklearn.cluster import KMeans
from sklearn.metrics import silhouette_score

random_state=1
k_range = list(range(2,11))
silhouette_scores_km=[]
for i in k_range:
    KMmodel = KMeans(n_clusters=i, random_state=random_state)
    y_km = KMmodel.fit_predict(X=X)
    silhouette_scores_km.append(silhouette_score(X=X, labels=y_km))

2.1 Consider carefully the number of clusters, simple optimisation of the silhouette will not be enough, decide visually the best number of clusters
